)
ADD_EXECUTABLE(TestVTKMeanImageFilter 
  
  TestVTKMeanImageFilter.cxx
  
  ${CMAKE_SOURCE_DIR}/${ORAIFUTILS_IMAGETOOLS_SUBDIR}/oraVTKMeanImageFilter.h
  ${CMAKE_SOURCE_DIR}/${ORAIFUTILS_IMAGETOOLS_SUBDIR}/oraVTKMeanImageFilter.cxx
)

  vtkCommon vtkIO vtkImaging
)

)

  COMMAND ${CMAKE_COMMAND} -E make_directory ${CMAKE_CURRENT_BINARY_DIR}/data
)

  COMMAND ${CMAKE_COMMAND} -E copy ${CMAKE_CURRENT_SOURCE_DIR}/data/2dimage.raw ${CMAKE_CURRENT_BINARY_DIR}/data 
)

  COMMAND ${CMAKE_COMMAND} -E copy ${CMAKE_CURRENT_SOURCE_DIR}/data/2dimage.mhd ${CMAKE_CURRENT_BINARY_DIR}/data
